# atlas/ingest/macro/fred_ingest.py
"""FRED API ingest for US 10Y, India 10Y, Brent USD, risk-free."""
from __future__ import annotations
import os
from decimal import Decimal
import pandas as pd
import requests
from sqlalchemy import text
from atlas.db import get_engine

FRED_BASE = "https://api.stlouisfed.org/fred/series/observations"
SERIES_MAP = {
    "us_10y_yield": "DGS10",
    "india_10y_yield": "INDIRLTLT01STM",
    "brent_usd": "DCOILBRENTEU",
    "risk_free_91d": "INDIRR1Y",  # proxy if no direct 91d series available
}

def fetch_series(series_id: str, start: str, end: str) -> pd.DataFrame:
    key = os.environ["FRED_API_KEY"]
    r = requests.get(FRED_BASE, params={
        "series_id": series_id, "api_key": key, "file_type": "json",
        "observation_start": start, "observation_end": end,
    }, timeout=30)
    r.raise_for_status()
    obs = r.json()["observations"]
    df = pd.DataFrame([{"date": o["date"], "value": float(o["value"])} for o in obs if o["value"] != "."])
    return df

def upsert_macro_col(col: str, df: pd.DataFrame) -> int:
    """UPSERT into atlas_macro_daily on date PK. Returns row count."""
    eng = get_engine()
    with eng.begin() as conn:
        for _, row in df.iterrows():
            conn.execute(text(f"""
                INSERT INTO atlas.atlas_macro_daily (date, {col}) VALUES (:d, :v)
                ON CONFLICT (date) DO UPDATE SET {col} = EXCLUDED.{col}
            """), {"d": row["date"], "v": Decimal(str(row["value"]))})
    return len(df)
